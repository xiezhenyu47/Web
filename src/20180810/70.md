class Solution {
 public int climbStairs(int n) {
        int last=1,lastlast=1,now=0;
        if(n==0||n==1){
            return 1;
        }else{
            for(int i=2;i<=n;i++){
                now=last+lastlast;
                lastlast=last;
                last=now;
            }
            return now;
        }
    }
}
